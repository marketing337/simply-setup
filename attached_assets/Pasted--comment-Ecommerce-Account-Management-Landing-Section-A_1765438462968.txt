      btn.addEventListener('click',()=>toggleQA(btn));
      btn.addEventListener('keydown',e=>{ if(e.key==='Enter'||e.key===' '){ e.preventDefault(); toggleQA(btn); }});
    });
    function toggleQA(btn){
      const qa = btn.parentElement;
      const open = qa.classList.contains('open');
      document.querySelectorAll('#eam-root .eam-qa.open').forEach(x=>{ x.classList.remove('open'); x.querySelector('.eam-q')?.setAttribute('aria-expanded','false'); });
      if(!open){ qa.classList.add('open'); btn.setAttribute('aria-expanded','true'); }
      else { qa.classList.remove('open'); btn.setAttribute('aria-expanded','false'); }
    }

    // Logos carousel
    (function(){
      const logoCarousel = document.querySelector('#eam-root #gst-logo-carousel');
      if(!logoCarousel) return;
      const originals = Array.from(document.querySelectorAll('#eam-root .gst-logo-slide'));
      originals.forEach(item=>logoCarousel.appendChild(item.cloneNode(true)));
      const SPEED_DESKTOP=120,SPEED_MOBILE=90; let pos=0,paused=false,last=performance.now();
      const isMobile=()=>window.innerWidth<=768;
      const speed=()=>window.matchMedia('(prefers-reduced-motion: reduce)').matches?0:(isMobile()?SPEED_MOBILE:SPEED_DESKTOP);
      const stride=()=>{const el=originals[0]; if(!el) return 0; const r=el.getBoundingClientRect(), cs=getComputedStyle(el); return r.width+parseFloat(cs.marginLeft)+parseFloat(cs.marginRight)};
      const total=()=>originals.length*stride();
      function tick(now){ const dt=(now-last)/1000; last=now; if(!paused){ pos-=speed()*dt; const tw=total(); if(tw>0 && pos<=-tw){ pos+=tw } logoCarousel.style.transform=`translateX(${pos}px)` } requestAnimationFrame(tick) }
      requestAnimationFrame(tick);
      logoCarousel.addEventListener('mouseenter',()=>paused=true);
      logoCarousel.addEventListener('mouseleave',()=>paused=false);
      window.addEventListener('resize',()=>{ const tw=total(); if(tw>0) pos=-((Math.abs(pos))%tw); last=performance.now(); logoCarousel.style.transform=`translateX(${pos}px)`; });
    })();
  </script>
